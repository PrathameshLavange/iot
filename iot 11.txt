#include <DHT.h>

#define DHTPIN 2
#define DHTTYPE DHT22
DHT dht(DHTPIN, DHTTYPE);

float minT = 1000, maxT = -1000;

void setup() {
  Serial.begin(9600);
  dht.begin();
}

void loop() {
  delay(2000);
  float t = dht.readTemperature();
  if (isnan(t)) {
    Serial.println("Sensor error!");
    return;
  }

  if (t < minT) minT = t;
  if (t > maxT) maxT = t;

  Serial.print("Temp: ");
  Serial.print(t);
  Serial.print(" Â°C | Min: ");
  Serial.print(minT);
  Serial.print(" | Max: ");
  Serial.println(maxT);
}
